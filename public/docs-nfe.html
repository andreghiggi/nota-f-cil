📄 API NF-e — Modelo 55 (Empresarial)

Base URL: https://vdzkhealunurfgrujekg.supabase.co/functions/v1/nfe-api

Guia completo de integração para emissão, consulta, cancelamento e reprocessamento de NF-e.

Autenticação: Todas as requisições exigem o header x-api-key: SEU_TOKEN_API. Gere tokens no painel em Tokens API.

1. Endpoints

MétodoEndpointDescrição
POST/nfe-apiEmitir nova NF-e
GET/nfe-api?status=...&limit=...Listar NF-e com filtros
GET/nfe-api/{id}Consultar NF-e por ID
POST/nfe-api/{id}/cancelarCancelar NF-e autorizada
POST/nfe-api/{id}/reprocessarReprocessar NF-e rejeitada
GET/nfe-api/{id}/xmlObter XMLs (envio/retorno)

Parâmetros de listagem (GET)

ParâmetroTipoDescrição

8. Integração Completa (Classe Client)

class NFeClient {
  constructor(apiKey, baseUrl) {
    this.apiKey = apiKey;
    this.baseUrl = baseUrl;
  }

  async request(endpoint, options = {}) {
    const response = await fetch(`${`this.baseUrl`}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': this.apiKey,
        ...options.headers
      }
    });
    const data = await response.json();
    if (!response.ok) throw new Error(data.error || 'API Error');
    return data;
  }

  async emitir(pedido) {
    return this.request('', {
      method: 'POST',
      body: JSON.stringify({
        external_id: pedido.id,
        natureza_operacao: pedido.natureza || 'VENDA',
        finalidade: pedido.finalidade || '1',
        modalidade_frete: pedido.frete_tipo || '9',
        destinatario: { /* dados do cliente */ },
        itens: pedido.itens.map(item => ({ /* mapeamento */ })),
        valor_desconto: pedido.desconto,
        valor_frete: pedido.frete
      })
    });
  }

  async consultar(nfeId) { return this.request(`/${nfeId}`); }
  async listar(filtros = {}) { return this.request(`?${new URLSearchParams(filtros)}`); }
  async cancelar(nfeId, justificativa) {
    return this.request(`/${nfeId}/cancelar`, {
      method: 'POST', body: JSON.stringify({ justificativa })
    });
  }
  async reprocessar(nfeId) { return this.request(`/${nfeId}/reprocessar`, { method: 'POST' }); }
  async obterXml(nfeId) { return this.request(`/${nfeId}/xml`); }
}

// Uso:
const client = new NFeClient('SEU_TOKEN', 'https://<BASE_URL>/functions/v1/nfe-api');
const resultado = await client.emitir(pedido);
console.log('NF-e criada:', resultado.data.id);

9. Webhooks NF-e

Eventos disponíveis

EventoDescrição
nfe.autorizadaNF-e autorizada pela SEFAZ
nfe.rejeitadaNF-e rejeitada pela SEFAZ
nfe.canceladaNF-e cancelada com sucesso
nfe.denegadaNF-e denegada pela SEFAZ

Validação de assinatura

const crypto = require('crypto');

function validateWebhookSignature(payload, signature, secret) {
  const expected = crypto
    .createHmac('sha256', secret)
    .update(JSON.stringify(payload))
    .digest('hex');
  return signature === `sha256=${expected}`;
}

app.post('/webhooks/fiscal', (req, res) => {
  const signature = req.headers['x-webhook-signature'];
  if (!validateWebhookSignature(req.body, signature, 'SEU_SECRET')) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  
  const { evento, dados } = req.body;
  switch (evento) {
    case 'nfe.autorizada':
      console.log(`NF-e ${dados.numero} autorizada!`);
      break;
    case 'nfe.rejeitada':
      console.log(`NF-e ${dados.numero} rejeitada: ${dados.motivo_retorno}`);
      break;
    case 'nfe.cancelada':
      console.log(`NF-e ${dados.numero} cancelada`);
      break;
  }
  
  res.status(200).json({ received: true });
});
